<?php

class fileReplaceArray {
    function doActionMultiple(array $replace){

       foreach(glob('/var/www/html/manuals/soumaia/*') as $path_to_file) {
         $file_contents   = file_get_contents($path_to_file);
         $new_file        = $path_to_file.'.swp' ;
         $keys            = array_keys($replace);
         $values          = array_values($replace);
         $replace_contents= str_replace($keys, $values, $file_contents);
         file_put_contents($new_file, $replace_contents);
       }

        return $replace_contents;
    }
}
